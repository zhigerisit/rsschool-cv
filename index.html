                function isPalindrome(line) {
                    let toS = line.toString();
                    let splitted = toS.split('');
                    let reverse = [];
                    for(let i = splitted.length -1; i >= 0; i--){
                        reverse.push(splitted[i]); 
                    }
                    let toJoin = reverse.join('');
                    if (toJoin == toS){
                        return true;
                    }
                    else {
                        return false;
                    }
                }
